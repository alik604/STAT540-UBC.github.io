X = function(A=1,B=-2,C=-1.5,D=-3,n=1000)
{
  dt = 1/n
  dX = (A + B*(1:n)*dt)*dt + exp( C + D*(1:n)*dt ) * sqrt(dt) * rnorm(n)
  return( cumsum(dX) )
}
library("graphics")
clr = rainbow(100)
